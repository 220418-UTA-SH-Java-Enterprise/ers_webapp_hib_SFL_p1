


create table ERS_REIMBURSEMENT_STATUS (
	id serial constraint  REIMBURSEMENT_STATUS_ID primary key,
	REIMBURSEMENT_STATUS varchar(25) not null
	
	
);

create table ERS_REIMBURSEMENT_TYPE (
	id serial constraint  REIMBURSEMENT_TYPE_ID primary key,
	REIMBURSEMENT_TYPE varchar(25) not null
	
	
);
create table ERS_USER_ROLES (
	id serial constraint  ERS_USER_ROLES_ID primary key,
	USER_ROLE varchar(25) not null
	
	
);




create table ERS_USERS (
	id serial constraint  ERS_USERS_ID primary key,
	USER_USERNAME varchar(25) not null,
	USER_PASSWORDE varchar(25) not null,
	USER_FIRST_NAME varchar(25) not null,
	USER_LAST_NAME varchar(25) not null,
	USER_ROLE_ID int not null
	
	
	
);

create table ERS_REIMBURSEMENT (
	id serial constraint  REIMB_ID primary key,
	REIMB_AMOUNT varchar(25) not null,
	REIMB_SUBMITTED date not null,
	REIMB_RESOLVED date not null,
	REIMB_DESCRIPTION varchar(25) not null,
	REIMB_AUTHOR int,
	REIMB_RESOLVER int,
	REIMB_STATUS_ID int,
	REIMB_TYPE_ID int

	
);
